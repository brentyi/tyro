3. Enums And Containers#

We can generate argument parsers from more advanced type annotations, like enums and container types.

 1import dataclasses
 2import enum
 3import pathlib
 4from typing import Optional, Tuple
 5
 6import dcargs
 7
 8
 9class OptimizerType(enum.Enum):
10    ADAM = enum.auto()
11    SGD = enum.auto()
12
13
14@dataclasses.dataclass(frozen=True)
15class TrainConfig:
16    # Example of a variable-length tuple. `typing.List`, `typing.Sequence`,
17    # `typing.Set`, `typing.Dict`, etc are all supported as well.
18    dataset_sources: Tuple[pathlib.Path, ...]
19    """Paths to load training data from. This can be multiple!"""
20
21    # Fixed-length tuples are also okay.
22    image_dimensions: Tuple[int, int] = (32, 32)
23    """Height and width of some image data."""
24
25    # Enums are handled seamlessly.
26    optimizer_type: OptimizerType = OptimizerType.ADAM
27    """Gradient-based optimizer to use."""
28
29    # We can also explicitly mark arguments as optional.
30    checkpoint_interval: Optional[int] = None
31    """Interval to save checkpoints at."""
32
33
34if __name__ == "__main__":
35    config = dcargs.cli(TrainConfig)
36    print(config)
