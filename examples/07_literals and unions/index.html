7. Literals And Unions#

typing.Literal[] can be used to restrict inputs to a fixed set of literal choices; typing.Union[] can be used to restrict inputs to a fixed set of types.

 1import dataclasses
 2import enum
 3from typing import Literal, Optional, Tuple, Union
 4
 5import dcargs
 6
 7
 8class Color(enum.Enum):
 9    RED = enum.auto()
10    GREEN = enum.auto()
11    BLUE = enum.auto()
12
13
14@dataclasses.dataclass(frozen=True)
15class Args:
16    # We can use Literal[] to restrict the set of allowable inputs, for example, over
17    # enums.
18    restricted_enum: Literal[Color.RED, Color.GREEN] = Color.RED
19
20    # Mixed types are okay.
21    mixed: Literal[Color.RED, Color.GREEN, "blue"] = "blue"
22
23    # Literals can also be marked Optional.
24    integer: Optional[Literal[0, 1, 2, 3]] = None
25
26    # Unions can be used to specify multiple allowable types.
27    union_over_types: Union[int, str] = 0
28    string_or_enum: Union[Literal["red", "green"], Color] = "red"
29
30    # Unions also work over more complex nested types.
31    union_over_tuples: Union[Tuple[int, int], Tuple[str]] = ("1",)
32
33    # And can be nested in other types.
34    tuple_of_string_or_enum: Tuple[Union[Literal["red", "green"], Color], ...] = (
35        "red",
36        Color.RED,
37    )
38
39
40if __name__ == "__main__":
41    args = dcargs.cli(Args)
42    print(args)
