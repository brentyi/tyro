6. Base Configs#

We can integrate dcargs.cli() into common configuration patterns: here, we select one of multiple possible base configurations, and then use the CLI to either override (existing) or fill in (missing) values.

Note that our interfaces don’t prescribe any of the mechanics used for storing or choosing between base configurations. A Hydra-style YAML approach could just as easily be used for the config libary (although we generally prefer to avoid YAMLs; staying in Python is convenient for autocompletion and type checking). For selection, we could also avoid fussing with sys.argv by using a BASE_CONFIG environment variable.

  1import sys
  2from dataclasses import dataclass
  3from typing import Callable, Dict, Literal, Tuple, TypeVar, Union
  4
  5from torch import nn
  6
  7import dcargs
  8
  9
 10@dataclass(frozen=True)
 11class AdamOptimizer:
 12    learning_rate: float = 1e-3
 13    betas: Tuple[float, float] = (0.9, 0.999)
 14
 15
 16@dataclass(frozen=True)
 17class SgdOptimizer:
 18    learning_rate: float = 3e-4
 19
 20
 21@dataclass(frozen=True)
 22class ExperimentConfig:
 23    # Dataset to run experiment on.
 24    dataset: Literal["mnist", "imagenet-50"]
 25
 26    # Optimizer parameters.
 27    optimizer: Union[AdamOptimizer, SgdOptimizer]
 28
 29    # Model size.
 30    num_layers: int
 31    units: int
 32
 33    # Batch size.
 34    batch_size: int
 35
 36    # Total number of training steps.
 37    train_steps: int
 38
 39    # Random seed. This is helpful for making sure that our experiments are all
 40    # reproducible!
 41    seed: int
 42
 43    # Activation to use. Not specifiable via the commandline.
 44    activation: Callable[[], nn.Module]
 45
 46
 47# Note that we could also define this library using separate YAML files (similar to
 48# `config_path`/`config_name` in Hydra), but staying in Python enables seamless type
 49# checking + IDE support.
 50base_configs = {
 51    "small": ExperimentConfig(
 52        dataset="mnist",
 53        optimizer=SgdOptimizer(),
 54        batch_size=2048,
 55        num_layers=4,
 56        units=64,
 57        train_steps=30_000,
 58        # The dcargs.MISSING sentinel allows us to specify that the seed should have no
 59        # default, and needs to be populated from the CLI.
 60        seed=dcargs.MISSING,
 61        activation=nn.ReLU,
 62    ),
 63    "big": ExperimentConfig(
 64        dataset="imagenet-50",
 65        optimizer=AdamOptimizer(),
 66        batch_size=32,
 67        num_layers=8,
 68        units=256,
 69        train_steps=100_000,
 70        seed=dcargs.MISSING,
 71        activation=nn.GELU,
 72    ),
 73}
 74
 75
 76T = TypeVar("T")
 77
 78
 79def cli_from_base_configs(base_library: Dict[str, T]) -> T:
 80    """Populate an instance of `cls`, where the first positional argument is used to
 81    select from a library of named base configs."""
 82    # Get base configuration name from the first positional argument.
 83    if len(sys.argv) < 2 or sys.argv[1] not in base_library:
 84        valid_usages = map(lambda k: f"{sys.argv[0]} {k} --help", base_library.keys())
 85        raise SystemExit("usage:\n  " + "\n  ".join(valid_usages))
 86
 87    # Get base configuration from our library, and use it for default CLI parameters.
 88    default_instance = base_library[sys.argv[1]]
 89    return dcargs.cli(
 90        type(default_instance),
 91        prog=" ".join(sys.argv[:2]),
 92        args=sys.argv[2:],
 93        default_instance=default_instance,
 94        # `avoid_subparsers` will avoid making a subparser for unions when a default is
 95        # provided; in this case, it simplifies our CLI but makes it less expressive
 96        # (cannot switch away from the base optimizer types).
 97        avoid_subparsers=True,
 98    )
 99
100
101if __name__ == "__main__":
102    config = cli_from_base_configs(base_configs)
103    print(config)
