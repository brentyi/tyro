16. Advanced Configuration#

The dcargs.conf module contains utilities that can be used to configure command-line interfaces beyond what is expressible via static type annotations.

Features here are supported, but generally unnecessary and should be used sparingly.

 1import dataclasses
 2from typing import Union
 3
 4from typing_extensions import Annotated
 5
 6import dcargs
 7
 8
 9@dataclasses.dataclass(frozen=True)
10class CheckoutArgs:
11    """Checkout a branch."""
12
13    branch: str
14
15
16@dataclasses.dataclass(frozen=True)
17class CommitArgs:
18    """Commit changes."""
19
20    message: str
21    all: bool = False
22
23
24@dataclasses.dataclass
25class Args:
26    # A boolean field with flag conversion turned off.
27    boolean: dcargs.conf.FlagConversionOff[bool] = False
28
29    # A numeric field parsed as a positional argument.
30    positional: dcargs.conf.Positional[int] = 3
31
32    # A numeric field that can't be changed via the CLI.
33    fixed: dcargs.conf.Fixed[int] = 5
34
35    # A union over nested structures, but without subcommand generation. When a default
36    # is provided, the type is simply fixed to that default.
37    union_without_subcommand: dcargs.conf.AvoidSubcommands[
38        Union[CheckoutArgs, CommitArgs]
39    ] = CheckoutArgs("main")
40
41    # `dcargs.conf.subcommand()` can be used to configure subcommands in a Union. Here,
42    # we make the subcommand names more succinct.
43    renamed_subcommand: Union[
44        Annotated[
45            CheckoutArgs, dcargs.conf.subcommand(name="checkout", prefix_name=False)
46        ],
47        Annotated[CommitArgs, dcargs.conf.subcommand(name="commit", prefix_name=False)],
48    ] = CheckoutArgs("main")
49
50
51if __name__ == "__main__":
52    print(dcargs.cli(Args))
